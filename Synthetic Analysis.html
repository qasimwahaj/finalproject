Synthetic Analysis

To simulate the data analysis, the following R chunk creates our synthetic data that incorporates the interrelationships among variables presented in the DAG.

Code 
# Step 1: Launching R packages we need
library(tidyverse)
library(scales)
library(truncnorm)
library(broom)
library(modelsummary)

# Step 2: Setting seeds for reproducability
set.seed(04212023)

# Step 3: Determining number of observations
n_people <- 1200

# Step 4: Creating exogenous variables in our model
my_data <- tibble(
  id = 1 : n_people,
  age = rtruncnorm(n_people, mean=28, sd = 9, a = 18, b = 50),
  num_relatives = rtruncnorm(n_people, mean=10, sd=5, a=2, b=50),
  legal_status = sample(c("Parolee", "SIV", "Asylee", "Refugee"), n_people,
                        replace = TRUE, prob = c(0.4, 0.3, 0.2, 0.1)),
  race = sample(c("Asian", "European", "African", "South American",
                  "other"), n_people, replace = TRUE, 
                prob = c(0.3, 0.2, 0.2, 0.2, 0.1)),
  education = rtruncnorm(n_people, mean = 12, sd = 3, a=0, b=25),
  male = sample(c(1, 0), n_people, replace = TRUE,
                prob = c(0.7, 0.3)),
  english_level = rtruncnorm(n_people, mean=70, sd = 8, a=65, b=100),
  employment = sample(c(1, 0), n_people, replace = TRUE,
                      prob = c(0.9, 0.1)),
  referral = sample(c(1, 0), n_people, replace = TRUE,
                    prob = c(0.95, 0.05)),
  time_since_arrival = runif(n_people, 0, 5),
  
# Step 5: Creating two stage variables
  pre_income_base = rtruncnorm(n_people, mean = 18, sd = 3, a=8, b = 35),
  job_satis_base = rtruncnorm(n_people, mean = 5, sd = 3, a = 1, b = 10))
            
my_data <- my_data %>% mutate(
  job_satis_emp = case_when(employment == 1 ~ 2,
                            employment == 0 ~ 0),
  job_satisfaction = job_satis_base + job_satis_emp,
  pre_income_emp = case_when(employment == 1 ~ 1,
                             employment == 0 ~ 0),
  pre_income = pre_income_base + pre_income_emp)

# Step 6: Creating the explanatory variable
my_data <- my_data %>% 
  mutate(program_baseline = sample(c(1, 0), n_people,
                                  replace = TRUE,prob = c(0.45, 0.55)),
         programe_age = age * 1.1,
         program_pre_income = pre_income * -1.11,
         program_social_network = num_relatives * -1.1,
         program_refugee_status = case_when(
           legal_status == "Parolee" ~ 5,
           legal_status == "SIV" ~ 3,
           legal_status == "Asylee" ~ 5,
           legal_status == "Refugee" ~ 2),
         program_race = case_when(
           race == "Asian" ~ 9,
           race == "European" ~ 5,
           race == "African" ~ 8,
           race == "South American" ~ 2,
           race == "other" ~ 3),
         program_education = education * -rnorm(n_people, 1.5, 1),
         program_sex = male * 3,
         program_job_satis = job_satisfaction * -4,
         program_english = english_level * 1.5,
         program_employment = employment * -20,
         program_referral = referral * 20,
         program_arrival = time_since_arrival * 2,
         program_prob = program_baseline + programe_age + program_pre_income +
           program_social_network + program_refugee_status + program_race +
           program_education + program_sex + program_job_satis + 
           program_english + program_employment + program_referral +
           program_arrival) %>% mutate(program_prob = rescale(program_prob, to = c(0, 1))) %>% mutate(program = rbinom(n_people, size = 1, prob = program_prob))

# Step 7: Creating the outcome variable
my_data <- my_data %>% 
  mutate(income_baseline = rtruncnorm(n_people, mean = 16, sd=2, a=8, b=30),
         income_age = age * 1.5,
         income_pre_income = pre_income * -1.1,
         income_social_network = num_relatives * 1.11,
         income_refugee_status = case_when(
           legal_status == "Parolee" ~ -3,
           legal_status == "SIV" ~ 5,
           legal_status == "Asylee" ~ -5,
           legal_status == "Refugee" ~ -6,
           legal_status == "No Legal Status" ~ 0),
         income_race = case_when(
           race == "Asian" ~ 5,
           race == "European" ~ 9,
           race == "African" ~ -5,
           race == "South American" ~ -6,
           race == "other" ~ 2),
         income_education = education * rnorm(n_people, 2, 2),
         income_sex = male * 5,
         income_job_satis = job_satisfaction * 2,
         income_english = english_level * 4,
         income_employment = employment * 4,
         income_referral = referral * 1,
         income_arrival = time_since_arrival * 2,
         income_program = program * 23,
         income = income_baseline + income_age + income_pre_income +
           income_social_network + income_refugee_status + income_race +
           income_education + income_sex + income_job_satis + 
           income_english + income_employment + income_referral +
           income_arrival + income_program)

# Step 8: Cleaning data by rescaling and rounding
my_data <- my_data %>% mutate(income = rescale(income, to = c(8, 30))) %>% 
  mutate(income = round(income, 2)) %>% 
  mutate(age = round(age, 0)) %>% mutate(time_since_arrival = round(time_since_arrival, 0)) %>% mutate(num_relatives = round(num_relatives, 0)) %>% mutate(education = round(education, 0)) %>% mutate(english_level = round(english_level, 2)) %>% mutate(pre_income = round(pre_income, 2)) %>% mutate(job_satisfaction = round(job_satisfaction, 0))

# Step 9: Creating a new dataset with only the variables we need for our analysis
data_neat <- my_data %>% select(id, age, male, pre_income, num_relatives,
                                 legal_status, race, education, employment,
                                 job_satisfaction, english_level, referral,
                                 time_since_arrival, program, income)

We use the synthetic that we just created and Inverse Probability Weighting (IPW) to estimate the impact of the program on dollar amount increase in income. The IPW method allows us to adjusting for the confounding factors that may affect enrollment in the program and the outcome. To do this, we assign a probability of receiving the program to each observation based on their characteristics in the confounding variables. In this method, the observations that were predicted to receive the program but did not, or vice versa, are given more weight than those that behave the way predicted, and that is how we adjust for confounders. The process of generating these weights involves two steps.

First, we generate propensity scores, which represent the probability of receiving the program.

Code 
model_logit <- glm(program ~ age + male + pre_income + num_relatives +
                   legal_status + race + education + employment +
                   job_satisfaction + english_level + referral +
                   time_since_arrival, 
                     data = data_neat,
                   family = binomial(link = "logit"))

program_probabilities <- augment_columns(model_logit,
                                     data_neat,
                                     type.predict = "response") %>%
  # The predictions are in a column named ".fitted", so we rename it here
  rename(propensity = .fitted)

Second, we use a specific formula to convert these scores into weights, giving more weight to observations that deviate from program enrollment based on propensity score.

Code 
program_ipw <- program_probabilities %>%
  mutate(ipw = (program / propensity) + ((1 - program) / (1 - propensity)))

By incorporating these inverse probability weights into our regression model, we can estimate the program's impact on dollar amount increase in predicted income.

Findings

As predicted, the result of our regression analysis suggest that participation in IRC’s Career Development Program leads to a statistically significant increase in predicted income leading to economic empowerment of refugee clients. Specifically, in our sample, program participation increases predicted income by $2 per hour on average. This effect can be generalized to the broader refugee population in the United States. At a 95% confidence level, we estimate that participation in IRC’s career development program increases refugee predicted income by between $1.7 and $2.4 per hour.

We can graphically show the statistical significance in the following plot:

Code 
plot_data <- tidy(model_ipw, conf.int = TRUE) %>% filter(term == "program")
ggplot(plot_data, aes(x = estimate, y = term)) + 
  geom_pointrange(aes(xmin = conf.low, xmax = conf.high), size = 2, linewidth = 1.7, color = "blue") +  geom_vline(xintercept = 0, color = "red", linetype = "dotted", size = 1.5) +
  theme_bw() +
  labs(x = "Dollar Increase in Hourly Income", y = "", title = "The Effect of the Program in Null World") +
  theme(plot.title = element_text(hjust = 0.5),
       plot.margin = unit(c(1.1,2,1.1,1), "cm")) +
 xlim(-.5, 3)

The plot indicates that the likelihood of obtaining the effect size of $2.07 is extremely unlikely in a scenario where there is no effect. The red dotted line serves as a reference for the scenario of no effect. Therefore, we can confidently concluded that the program has a significant effect on increasing income.
